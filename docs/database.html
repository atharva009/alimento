222    def save_analysis(self, analysis_data):
223        """Persist a meal analysis document to the ``analysis_history`` collection.
224
225        Automatically sets ``timestamp`` (ISO string) and ``created_at``
226        (datetime) if absent.  Ownership fields ``user_id`` and
227        ``guest_session_id`` default to ``None`` if not supplied.
228
229        Args:
230            analysis_data: Dict containing the analysis payload.  Modified
231                           in-place to add metadata fields before insertion.
232
233        Returns:
234            ``{"success": True, "id": "<ObjectId str>"}`` on success, or
235            ``{"success": False, "error": "<message>"}`` on failure.
236        """
237        if not self.client:
238            return {"success": False, "error": "Database not connected"}
239        
240        try:
241            # Add timestamp if not present (UTC)
242            if 'timestamp' not in analysis_data:
243                analysis_data['timestamp'] = datetime.utcnow().isoformat()
244            
245            # Add created_at for sorting (UTC)
246            analysis_data['created_at'] = datetime.utcnow()
247
248            # Ensure ownership fields exist (nullable)
249            analysis_data.setdefault('user_id', None)
250            analysis_data.setdefault('guest_session_id', None)
251            
252            print(f"Attempting to save analysis to MongoDB...")
253            
254            # Insert document
255            result = self.collection.insert_one(analysis_data)
256            
257            print(f"Analysis saved with ID: {result.inserted_id}")
258            return {"success": True, "id": str(result.inserted_id)}
259            
260        except Exception as e:
261            print(f"Save error: {e}")
262            print(f"Error type: {type(e).__name__}")
263            return {"success": False, "error": str(e)}

265    def get_history(self, limit=20):
266        """Retrieve recent meal analyses sorted newest-first.
267
268        Args:
269            limit: Maximum number of documents to return (default ``20``).
270
271        Returns:
272            List of analysis dicts with ``_id`` converted to string and a
273            ``timestamp`` ISO string derived from ``created_at``.  Returns
274            ``[]`` if the database is not connected or on error.
275        """
276        if not self.client:
277            print("Database not connected, returning empty history")
278            return []
279        
280        try:
281            print(f"Attempting to retrieve {limit} analyses...")
282            
283            # Get documents sorted by created_at (newest first)
284            cursor = self.collection.find().sort("created_at", -1).limit(limit)
285            
286            # Convert to list and handle ObjectId serialization
287            history = []
288            for doc in cursor:
289                # Convert ObjectId to string for JSON serialization
290                doc['_id'] = str(doc['_id'])
291                
292                # Ensure timestamp is in the format expected by templates
293                if 'created_at' in doc:
294                    doc['timestamp'] = doc['created_at'].isoformat()
295                
296                history.append(doc)
297            
298            print(f"Retrieved {len(history)} analyses from database")
299            return history
300            
301        except Exception as e:
302            print(f"History retrieval error: {e}")
303            print(f"Error type: {type(e).__name__}")
304            return []

358        """Return basic database and collection statistics.
359
360        Returns:
361            Dict containing at minimum ``total_analyses`` (int) and
362            ``connected`` (bool).  When data exists, also includes
363            ``database_size`` and ``collection_size`` in bytes if the
364            server supports ``dbStats`` / ``collStats`` commands.
365            Returns ``{"error": "..."}`` when the database is not connected.
366        """
367        if not self.client:
368            return {"error": "Database not connected"}
369        
370        try:
371            total_analyses = self.collection.count_documents({})
372            
373            stats = {
374                "total_analyses": total_analyses,
375                "connected": True
376            }
377            
378            # Only get advanced stats if we have data
379            if total_analyses > 0:
380                try:
381                    db_stats = self.db.command("dbStats")
382                    stats["database_size"] = db_stats.get("dataSize", 0)
383                    
384                    coll_stats = self.db.command("collStats", "analysis_history")
385                    stats["collection_size"] = coll_stats.get("size", 0)
386                except:
387                    # Advanced stats failed, but basic stats work
388                    pass
389            
390            return stats
391            
392        except Exception as e:
393            print(f"Stats error: {e}")
394            return {"error": str(e), "connected": False}

Singleton-style wrapper around a PyMongo MongoClient.

Manages the connection lifecycle, collection references, and index creation for all Alimento collections in MongoDB Atlas. When MONGODB_URI is absent the instance degrades gracefully: client and db are None and all data-access methods return safe fallback values instead of raising.

Collections exposed as attributes (each is a pymongo.Collection):

  • collection – legacy analysis_history
  • users, logins, usage, share_links, hydration_logs
  • meal_logs, food_items, inventory_items, barcode_cache
  • recipes, meal_plans, grocery_lists, weight_logs
  • chat_sessions, challenges, challenge_members
  • activity_integrations, notification_settings, migration_state
  • user_profiles, nutrition_goals, diet_preferences

Environment variables:

  • MONGODB_URI – MongoDB Atlas connection string (required).
  • AUTO_CREATE_INDEXES – Set to '1' to run ensure_indexes() at startup; omit for faster cold starts (indexes created on demand).
